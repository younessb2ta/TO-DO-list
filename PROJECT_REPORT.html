Database Schema

CREATE TABLE tasks (
  id INT AUTO_INCREMENT PRIMARY KEY,
  task_text VARCHAR(255) NOT NULL,
  due_date DATETIME,
  completed BOOLEAN DEFAULT FALSE,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
